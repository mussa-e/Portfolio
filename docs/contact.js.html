/**
 * Shows or hides a validation error message.
 * @function
 * @param {string} id - ID of the error element
 * @param {boolean} valid - Indicates whether the field is valid
 * @returns {void}
 */
function toggleError(id, valid) {
    const error = document.getElementById(id);
    error.classList.toggle("d-none", valid);
}


/**
 * Validates the name input field.
 * @function
 * @returns {boolean} True if the name is valid, otherwise false
 */
function validateName() {
    const value = document
        .getElementById("name")
        .value
        .trim();
    const valid = value.length >= 2;
    toggleError("val-name", valid);
    
    return valid;
}


/**
 * Validates the email input field.
 * @function
 * @returns {boolean} True if the email is valid, otherwise false
 */
function validateEmail() {
    const value = document
        .getElementById("email")
        .value
        .trim();
    const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    const valid = regex.test(value);
    toggleError("val-mail", valid);
    
    return valid;
}


/**
 * Validates the message input field.
 * @function
 * @returns {boolean} True if the message is valid, otherwise false
 */
function validateMessage() {
    const value = document.getElementById("message").value.trim();
    const valid = value.length >= 5;
    toggleError("val-message", valid);
    
    return valid;
}


/**
 * Validates whether the user has agreed to the terms.
 * @function
 * @returns {boolean} True if the terms are accepted, otherwise false
 */
function validateTerms() {
    toggleError("val-terms", agreedTerms);
    return agreedTerms;
}


/**
 * Toggles the terms agreement checkbox state and updates the UI.
 * @function
 * @returns {void}
 */
function toggleAgree() {

    let checkBox = document.getElementById("agree-img");

    if (checkBox.src.includes("check-box.svg")) {
        checkBox.src = "img/haekchen4.png";
        agreedTerms = true;
    } else {
        checkBox.src = "img/check-box.svg";
        agreedTerms = false;
    }

    validateTerms();
    checkSendBtn();
}

/**
 * Handles the form submission process.
 * @async
 * @function
 * @param {SubmitEvent} e - Form submit event
 * @returns {Promise<void>}
 */
async function handleSubmit(e) {
    e.preventDefault();

    if (!isFormValid()) return;

    try {
        const result = await sendFormData(getFormData());
        handleSubmitResult(result);
    } catch (error) {
        handleSubmitError(error);
    }
}


/**
 * Checks whether all form fields are valid.
 * @function
 * @returns {boolean} True if the form is valid, otherwise false
 */
function isFormValid() {
    return (
        validateName() &&
        validateEmail() &&
        validateMessage() &&
        validateTerms()
    );
}


/**
 * Collects all form data.
 * @function
 * @returns {{name: string, email: string, message: string}} Form data object
 */
function getFormData() {
    return {
        name: getValue("name"),
        email: getValue("email"),
        message: getValue("message")
    };
}


/**
 * Returns the value of an input element.
 * @function
 * @param {string} id - ID of the input element
 * @returns {string} Input value
 */
function getValue(id) {
    return document.getElementById(id).value;
}


/**
 * Sends form data to the server.
 * @async
 * @function
 * @param {{name: string, email: string, message: string}} data - Form data to send
 * @returns {Promise<Object>} Server response
 */
async function sendFormData(data) {
    const response = await fetch("/contact.php", {
        method: "POST",
        headers: {
            "Content-Type": "application/json"
        },
        body: JSON.stringify(data)
    });

    return response.json();
}


/**
 * Processes the server response after form submission.
 * @function
 * @param {Object} result - Response object from the server
 * @returns {void}
 */
function handleSubmitResult(result) {
    if (result.success) {
        resetFormState();
        messageSendFeedback("succeeded");
        return;
    }

    alert("Fehler: " + result.error);
}


/**
 * Resets the form and agreement state.
 * @function
 * @returns {void}
 */
function resetFormState() {
    form.reset();
    agreedTerms = false;
}


/**
 * Handles errors that occur during form submission.
 * @function
 * @param {Error} error - Error object
 * @returns {void}
 */
function handleSubmitError(error) {
    messageSendFeedback("failed");
    console.error(error);
}


/**
 * Enables or disables the send button depending on form validity.
 * @function
 * @returns {void}
 */
function checkSendBtn() {
    let btn = document.getElementById("send-btn");

    const formValid =
        validateName() &&
        validateEmail() &&
        validateMessage() &&
        agreedTerms;

    if (formValid) {
        buttonActive(btn);
    } else {
        buttonNotActive(btn);
    }
}


/**
 * Activates the submit button.
 * @function
 * @param {HTMLButtonElement} btn - Submit button element
 * @returns {void}
 */
function buttonActive(btn){
    btn.classList.remove("not-active");
    btn.classList.add("formular-send");
    btn.disabled = false;
}


/**
 * Deactivates the submit button.
 * @function
 * @param {HTMLButtonElement} btn - Submit button element
 * @returns {void}
 */
function buttonNotActive(btn){
    btn.classList.add("not-active");
    btn.classList.remove("formular-send");
    btn.disabled = true;
}


/**
 * Displays a feedback message after form submission.
 * @function
 * @param {string} value - Feedback status ("succeeded" or "failed")
 * @returns {void}
 */
function messageSendFeedback(value) {
    const feedback = document.getElementById("feedback");
    feedback.innerHTML = "";

    if(value === "succeeded"){
        messageSuccesful(feedback);
    }
    if(value === "failed"){
        messageFailed(feedback);
    }

    setTimeout(() => {
        feedback.classList.remove("show-success");
        feedback.classList.remove("show-failed");
        feedback.classList.add("hidden");
    }, 2500);
}


/**
 * Displays a success feedback message.
 * @function
 * @param {HTMLElement} feedback - Feedback container element
 * @returns {void}
 */
function messageSuccesful(feedback){
        feedback.innerHTML = "Message sent succesfully";
        feedback.classList.remove("hidden");
        feedback.classList.add("show-success");
}


/**
 * Displays an error feedback message.
 * @function
 * @param {HTMLElement} feedback - Feedback container element
 * @returns {void}
 */
function messageFailed(feedback){
        feedback.innerHTML = "Message could not be sent";
        feedback.classList.remove("hidden");
        feedback.classList.add("show-failed");
}
