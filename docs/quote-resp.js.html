/**
 * Initializes the quotes slider functionality after the DOM is loaded.
 * Sets up horizontal scrolling and observes quote visibility to update steppers.
 *
 * @function
 * @returns {void}
 */
function initQuotesSlider() {
    const quotesWrapper = document.querySelector('.quotes-wrapper');
    const quotes = [...document.querySelectorAll('.quotes')];
    const steppers = getSteppers();

    enableHorizontalScroll(quotesWrapper);
    observeQuotes(quotesWrapper, quotes, steppers);
}


/**
 * Retrieves all stepper elements used for navigation.
 *
 * @function
 * @returns {HTMLElement[]} Array of stepper elements
 */
function getSteppers() {
    return [
        document.querySelector('.stepper-1'),
        document.querySelector('.stepper-2'),
        document.querySelector('.stepper-3')
    ];
}


/**
 * Enables horizontal scrolling with the mouse wheel on the given wrapper element.
 *
 * @function
 * @param {HTMLElement} wrapper - The container element that should scroll horizontally
 * @returns {void}
 */
function enableHorizontalScroll(wrapper) {
    wrapper.addEventListener(
        'wheel',
        (event) => handleWheelScroll(event, wrapper),
        { passive: false }
    );
}


/**
 * Handles mouse wheel scrolling and converts vertical wheel movement
 * into smooth horizontal scrolling.
 *
 * @function
 * @param {WheelEvent} event - The wheel event triggered by the user
 * @param {HTMLElement} wrapper - The container element to scroll
 * @returns {void}
 */
function handleWheelScroll(event, wrapper) {
    event.preventDefault();

    wrapper.scrollBy({
        left: event.deltaY * 0.4,
        behavior: 'smooth'
    });
}


/**
 * Creates an observer and starts observing all quote elements.
 *
 * @function
 * @param {HTMLElement} wrapper - The scrolling container element
 * @param {HTMLElement[]} quotes - Array of quote elements
 * @param {HTMLElement[]} steppers - Array of stepper elements
 * @returns {void}
 */
function observeQuotes(wrapper, quotes, steppers) {
    const observer = createObserver(wrapper, quotes, steppers);

    quotes.forEach((quote) => {
        observer.observe(quote);
    });
}


/**
 * Creates an IntersectionObserver to track which quote is currently visible.
 *
 * @function
 * @param {HTMLElement} wrapper - The scrolling container element
 * @param {HTMLElement[]} quotes - Array of quote elements
 * @param {HTMLElement[]} steppers - Array of stepper elements
 * @returns {IntersectionObserver} Configured IntersectionObserver instance
 */
function createObserver(wrapper, quotes, steppers) {
    return new IntersectionObserver(
        (entries) => handleIntersections(entries, quotes, steppers),
        {
            root: wrapper,
            threshold: 0.6
        }
    );
}


/**
 * Handles intersection changes and updates the active stepper
 * when a quote becomes visible.
 *
 * @function
 * @param {IntersectionObserverEntry[]} entries - Array of intersection entries
 * @param {HTMLElement[]} quotes - Array of quote elements
 * @param {HTMLElement[]} steppers - Array of stepper elements
 * @returns {void}
 */
function handleIntersections(entries, quotes, steppers) {
    entries.forEach((entry) => {
        if (!entry.isIntersecting) {
            return;
        }

        updateActiveStepper(entry.target, quotes, steppers);
    });
}


/**
 * Updates the active stepper based on the currently visible quote.
 *
 * @function
 * @param {Element} target - The currently visible quote element
 * @param {HTMLElement[]} quotes - Array of quote elements
 * @param {HTMLElement[]} steppers - Array of stepper elements
 * @returns {void}
 */
function updateActiveStepper(target, quotes, steppers) {
    steppers.forEach((stepper) => {
        stepper.classList.remove('active');
    });

    const index = quotes.indexOf(target);

    steppers[index]?.classList.add('active');
}


/**
 * Scrolls smoothly to the selected quote based on the provided choice.
 *
 * @function
 * @param {string} choice - The quote identifier ("one", "two", or "three")
 * @returns {void}
 */
function quoteChoice(choice) {
    const map = {
        one: 0,
        two: 1,
        three: 2
    };

    const quotes = document.querySelectorAll('.quotes');

    quotes[map[choice]].scrollIntoView({
        behavior: 'smooth',
        inline: 'center',
        block: 'nearest'
    });
}
